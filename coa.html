function saveAccount(){ let code = document.getElementById("code").value; let name = document.getElementById("accountName").value; if(!code || !name){ alert("Please fill all fields"); return; } let accounts = JSON.parse(localStorage.getItem("accounts")) || []; accounts.push({ code: code, name: name }); localStorage.setItem("accounts", JSON.stringify(accounts)); alert("Saved!"); }
